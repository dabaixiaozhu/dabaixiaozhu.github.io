# <a name="1">[1 前缀树](#begin)</a>

## <a name="1.1">[概念](#begin)</a>

- 1 单个字符串中，字符从前到后的加到一棵多叉树上
- 2 字符放在路上，节点上有专属的数据项（常见的是pass和end值）
- 3 所有样本都这样添加，如果没有路就新建，如有路就复用
- 4 沿途节点的pass值增加1，每个字符串结束时来到的节点end值增加1

## <a name="1.2">[实现1](#begin)</a>

### <a name="1.21">[1 添加某个字符串，可以重复添加，每次算1个](#begin)</a>

```java
// 1 加入一个word
public void insert(String word) {
    if (word == null) {
        return;
    }
    char[] str = word.toCharArray();
    Node1 node = root;
    // 要新加字符串 头节点的pass先++
    node.pass++;
    int path = 0;
    // 遍历传入的字符串中的字符
    for (int i = 0; i < str.length; i++) { // 从左往右遍历字符
        // 由字符，对应成走向哪条路 a即0的位置的路
        path = str[i] - 'a';
        // 当前这条路下一个节点是否为null
        if (node.nexts[path] == null) {
            // 在此方向上建立新节点
            node.nexts[path] = new Node1();
        }
        // 跳到下一个节点
        node = node.nexts[path];
        // 只要到了下个节点 pass++
        node.pass++;
    }
    // 遍历完一个字符串，这里的node是最后一个节点了 end++
    node.end++;
}
```

#### 代码

```java
// 3 删除路径
public void delete(String word) {
    // 必须要有此路径 才能删除
    if (search(word) != 0) {
        char[] chs = word.toCharArray();
        // 头节点
        Node1 node = root;
        // 头节点的p先要-1
        node.pass--;
        int path = 0;
        for (int i = 0; i < chs.length; i++) {
            // 下一级的路
            path = chs[i] - 'a';
            // 下一级节点p值-- 后为0
            if (--node.nexts[path].pass == 0) {
                // 设置此节点为空 直接return下面的内容不要了
                node.nexts[path] = null;
                return;
            }
            // 下一个节点
            node = node.nexts[path];
        }
        node.end--;
    }
}
```

### <a name="1.25">[整体代码](#begin)</a>

```java
// 实现1
public static class Node1 {
    public int pass;
    public int end;
    public Node1[] nexts;

    public Node1() {
        pass = 0;
        end = 0;
        // 每个节点下一条路 最多就是从26个字母中选一个
        // 0    a
        // 1    b
        // 2    cd
        // ..   ..
        // 25   z
        // nexts[i] == null   i方向的路不存在
        // nexts[i] != null   i方向的路存在
        nexts = new Node1[26];
    }
}

public static class Trie1 {
    private Node1 root;

    // 初始化只需要头节点
    public Trie1() {
        // 初始化的节点 p e默认为0
        root = new Node1();
    }

    // 1 加入一个word
    public void insert(String word) {
        if (word == null) {
            return;
        }
        char[] str = word.toCharArray();
        Node1 node = root;
        // 要新加字符串 头节点的pass先++
        node.pass++;
        int path = 0;
        // 遍历传入的字符串中的字符
        for (int i = 0; i < str.length; i++) { // 从左往右遍历字符
            // 由字符，对应成走向哪条路 a即0的位置的路
            path = str[i] - 'a';
            // 当前这条路下一个节点是否为null
            if (node.nexts[path] == null) {
                // 在此方向上建立新节点
                node.nexts[path] = new Node1();
            }
            // 跳到下一个节点
            node = node.nexts[path];
            // 只要到了下个节点 pass++
            node.pass++;
        }
        // 遍历完一个字符串，这里的node是最后一个节点了 end++
        node.end++;
    }

    // 2 word这个单词之前加入过几次
    // 这里 p e值不用增加
    public int search(String word) {
        // 不可能加入为空
        if (word == null) {
            return 0;
        }
        char[] chs = word.toCharArray();
        Node1 node = root;
        int index = 0;
        for (int i = 0; i < chs.length; i++) {
            index = chs[i] - 'a';
            // 下标没路了 表示到了最后一个节点
            if (node.nexts[index] == null) {
                return 0;
            }
            node = node.nexts[index];
        }
        // 最后一个节点的end值，就是加入过的次数
        return node.end;
    }

    // 3 删除路径
    public void delete(String word) {
        // 必须要有此路径 才能删除
        if (search(word) != 0) {
            char[] chs = word.toCharArray();
            // 头节点
            Node1 node = root;
            // 头节点的p先要-1
            node.pass--;
            int path = 0;
            for (int i = 0; i < chs.length; i++) {
                // 下一级的路
                path = chs[i] - 'a';
                // 下一级节点p值-- 后为0
                if (--node.nexts[path].pass == 0) {
                    // 设置此节点为空 直接return下面的内容不要了
                    node.nexts[path] = null;
                    return;
                }
                // 下一个节点
                node = node.nexts[path];
            }
            node.end--;
        }
    }

    // 4 所有加入的字符串中，有几个是以pre这个字符串作为前缀的
    public int prefixNumber(String pre) {
        if (pre == null) {
            return 0;
        }
        char[] chs = pre.toCharArray();
        Node1 node = root;
        int index = 0;
        for (int i = 0; i < chs.length; i++) {
            index = chs[i] - 'a';
            // 不存在此路径
            if (node.nexts[index] == null) {
                return 0;
            }
            node = node.nexts[index];
        }
        // p值则为经过值
        return node.pass;
    }
}
```

## <a name="1.3">[实现2 hashMap<index,Node2>](#begin)</a>

同样包括 实现1 的四个方法

```java
// 实现2
// 实现1只是字母 如果输入的字符种类很多，就要用哈希表的形式
public static class Node2 {
    public int pass;
    public int end;
    // 哈希表 其余的没有区别
    public HashMap<Integer, Node2> nexts;

    public Node2() {
        pass = 0;
        end = 0;
        nexts = new HashMap<>();
    }
}

public static class Trie2 {
    private Node2 root;

    // 初始化只需要头节点
    public Trie2() {
        root = new Node2();
    }

    // 加入一个word
    public void insert(String word) {
        if (word == null) {
            return;
        }
        char[] chs = word.toCharArray();
        Node2 node = root;
        node.pass++;
        int index = 0;
        for (int i = 0; i < chs.length; i++) {
            // 把字符转为数字
            index = (int) chs[i];
            // 下一个节点是否为空 即map中是否含有此key
            if (!node.nexts.containsKey(index)) {
                // 如果节点为空 则新建节点
                node.nexts.put(index, new Node2());
            }
            // 跳到下一个节点
            node = node.nexts.get(index);
            // 只要到了下个节点 pass++
            node.pass++;
        }
        node.end++;
    }

    // word这个单词之前加入过几次
    // 这里 p e值不用增加
    public int search(String word) {
        if (word == null) {
            return 0;
        }
        char[] chs = word.toCharArray();
        Node2 node = root;
        int index = 0;
        for (int i = 0; i < chs.length; i++) {
            index = (int) chs[i];
            if (!node.nexts.containsKey(index)) {
                return 0;
            }
            node = node.nexts.get(index);
        }
        // 最后一个节点的end值，就是加入过的次数
        return node.end;
    }

    // 删除路径
    public void delete(String word) {
        // 必须要有此路径 才能删除
        if (search(word) != 0) {
            char[] chs = word.toCharArray();
            // 头节点
            Node2 node = root;
            // 头节点的p先要-1
            node.pass--;
            int index = 0;
            for (int i = 0; i < chs.length; i++) {
                index = (int) chs[i];
                // 下一级节点p值-- 后为0
                if (--node.nexts.get(index).pass == 0) {
                    // 删除此节点 直接return下面的内容不要了
                    node.nexts.remove(index);
                    return;
                }
                // 下一个节点的值
                node = node.nexts.get(index);
            }
            node.end--;
        }
    }

    // 所有加入的字符串中，有几个是以pre这个字符串作为前缀的
    public int prefixNumber(String pre) {
        if (pre == null) {
            return 0;
        }
        char[] chs = pre.toCharArray();
        Node2 node = root;
        int index = 0;
        for (int i = 0; i < chs.length; i++) {
            index = (int) chs[i];
            // 不存在此路径
            if (!node.nexts.containsKey(index)) {
                return 0;
            }
            node = node.nexts.get(index);
        }
        // p值则为经过值
        return node.pass;
    }
}
```
