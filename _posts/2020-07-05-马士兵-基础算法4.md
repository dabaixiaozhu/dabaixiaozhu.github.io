#### <a name="1.21">[heapInsert](#begin)</a>

```java
// 与父树值比较，上滑
private void heapInsert(int[] arr, int index) {
    // index位置的值 是新插入的数
    // 插入值>父节点值 index来到0位置，即来到顶部 会停止
    while (arr[index] > arr[(index - 1) / 2]) {
        // 值交换
        swap(arr, index, (index - 1) / 2);
        // 坐标交换
        index = (index - 1) / 2;
    }
}

private void swap(int[] arr, int i, int j) {
    int tmp = arr[i];
    arr[i] = arr[j];
    arr[j] = tmp;
}
```

#### <a name="1.22">[heapify](#begin)</a>

```java
// 与子树值比较：从index位置，往下看 不断下沉
// 停止条件：子树<=值 或者没有子树了
private void heapify(int[] arr, int index, int heapSize) {
    // 左子树位置
    int left = index * 2 + 1;
    // 代表有子树
    while (left < heapSize) {
        // 左右子树 谁的值大 返回哪个值
        int largest = left + 1 < heapSize && arr[left + 1] > arr[left] ? left + 1 : left;
        // 左右子树中值较大的 根index的值进行比较 谁大返回哪个下标
        largest = arr[largest] > arr[index] ? largest : index;
        // index的值比左右子树的值大  不需要下沉了
        if (largest == index) {
            break;
        }
        // 交换值
        swap(arr, largest, index);
        // index下沉
        index = largest;
        // 新的index的左子树下标
        left = index * 2 + 1;
    }
}
```

// 二叉堆
public static class MyMaxHeap {
    private int[] heap;
    private final int limit;
    private int heapSize;

    public MyMaxHeap(int limit) {
        heap = new int[limit];
        this.limit = limit;
        heapSize = 0;
    }

    public boolean isEmpty() {
        return heapSize == 0;
    }

    public boolean isFull() {
        return heapSize == limit;
    }

    // 与父树值比较，上滑
    private void heapInsert(int[] arr, int index) {
        // index位置的值 是新插入的数
        // 插入值>父节点值 index来到0位置，即来到顶部 会停止
        while (arr[index] > arr[(index - 1) / 2]) {
            // 值交换
            swap(arr, index, (index - 1) / 2);
            // 坐标交换
            index = (index - 1) / 2;
        }
    }

    // 与子树值比较：从index位置，往下看 不断下沉
    // 停止条件：子树<=值 或者没有子树了
    private void heapify(int[] arr, int index, int heapSize) {
        // 左子树位置
        int left = index * 2 + 1;
        // 代表有子树
        while (left < heapSize) {
            // 左右子树 谁的值大 返回哪个值
            int largest = left + 1 < heapSize && arr[left + 1] > arr[left] ? left + 1 : left;
            // 左右子树中值较大的 根index的值进行比较 谁大返回哪个下标
            largest = arr[largest] > arr[index] ? largest : index;
            // index的值比左右子树的值大  不需要下沉了
            if (largest == index) {
                break;
            }
            // 交换值
            swap(arr, largest, index);
            // index下沉
            index = largest;
            // 新的index的左子树下标
            left = index * 2 + 1;
        }
    }

    private void swap(int[] arr, int i, int j) {
        int tmp = arr[i];
        arr[i] = arr[j];
        arr[j] = tmp;
    }

    // 插入数据
    public void push(int value) {
        if (heapSize == limit) {
            throw new RuntimeException("heap is full");
        }
        // 新进来的数放置的位置 新插入
        heap[heapSize] = value;
        heapInsert(heap, heapSize++);
    }

    // 返回并删除最大值 剩下的数依然是大根堆
    public int pop() {
        // 最大值
        int ans = heap[0];
        // 最后一个值与头位置值交换 heapSize再-1（去掉堆的最后一个值）
        swap(heap, 0, --heapSize);
        // 头节点是新值 跟子节点进行比较
        heapify(heap, 0, heapSize);
        return ans;
    }
}

### <a name="1.255">[暴力方法](#begin)</a>

```java
// 暴力方法
public static class RightMaxHeap {
    private int[] arr;
    private final int limit;
    private int size;

    public RightMaxHeap(int limit) {
        arr = new int[limit];
        this.limit = limit;
        size = 0;
    }

    public boolean isEmpty() {
        return size == 0;
    }

    public boolean isFull() {
        return size == limit;
    }

    // 添加元素
    public void push(int value) {
        if (size == limit) {
            throw new RuntimeException("heap is full");
        }
        arr[size++] = value;
    }

    // 弹出元素 只是弹出最大值
    public int pop() {
        int maxIndex = 0;
        for (int i = 1; i < size; i++) {
            if (arr[i] > arr[maxIndex]) {
                maxIndex = i;
            }
        }
        int ans = arr[maxIndex];
        // 重新赋值 数组长度-1
        arr[maxIndex] = arr[--size];
        return ans;
    }
}
```

## <a name="1.3">[堆排序](#begin)</a>

### <a name="1.31">[思路](#begin)</a>

- 1 先让整个数组都变成大根堆结构，建立堆的过程:
  - 1)从上到下的方法，时间复杂度为O(N*logN)
  - 2)从下到上的方法，时间复杂度为O(N)
- 2 把堆的最大值和堆末尾的值交换，然后减少堆的大小之后，再去调整堆，一直周而复始，时间复杂度为O(N*logN)
- 3 堆的大小减小成0之后，排序完成

### <a name="1.33">[经典堆排序](#begin)</a>

#### 思路

heapInsert：从上往下 O(NlogN)

![6](https://dabaixiaozhu.oss-cn-shenzhen.aliyuncs.com/LinBlog/mashibing/basicStructure/class4/stack/6.png)

#### 代码

```java
// 堆排序额外空间复杂度O(1) 没有使用递归
public static void heapSort(int[] arr) {
    if (arr == null || arr.length < 2) {
        return;
    }
    // O(N*logN) 经典堆排序
    // 假设数组中每个数都是 新给的 来一个heapInsert
    for (int i = 0; i < arr.length; i++) { // O(N)
        heapInsert(arr, i); // O(logN)
    }

    int heapSize = arr.length;
    // 0位置的数 跟heapSize上的数交换后 最后一位不再变化
    // 此时最大值在最后一位
    swap(arr, 0, --heapSize);
    // O(N*logN)
    while (heapSize > 0) { // O(N)
        // 再把减去最后一位的arr 重新调整为大根堆
        heapify(arr, 0, heapSize); // O(logN)
        // 交换值 最后一位不再变化
        swap(arr, 0, --heapSize); // O(1)
    }
}

// heapInsert如上
```

#### 思路

```java
// 堆排序额外空间复杂度O(1) 没有使用递归
public static void heapSort(int[] arr) {
    if (arr == null || arr.length < 2) {
        return;
    }

    // 优化 一股脑把数组内容给过来 不是一个个新添 成为大根堆
    // 优化为O(N)
    for (int i = arr.length - 1; i >= 0; i--) {
        heapify(arr, i, arr.length);
    }

    int heapSize = arr.length;
    // 0位置的数 跟heapSize上的数交换后 最后一位不再变化
    // 此时最大值在最后一位
    swap(arr, 0, --heapSize);
    // O(N*logN)
    while (heapSize > 0) { // O(N)
        // 再把减去最后一位的arr 重新调整为大根堆
        heapify(arr, 0, heapSize); // O(logN)
        // 交换值 最后一位不再变化
        swap(arr, 0, --heapSize); // O(1)
    }
}

// heapify如上
```

### <a name="1.42">[一个几乎有序的数组，选择一个合适的排序策略，对这个数组进行排序](#begin)</a>

#### 几乎有序

如果把数组排好顺序的话，每个元素移动的距离一定不超过k，并且k相对于数组长度来说是比较小的

#### 思路

![9](https://dabaixiaozhu.oss-cn-shenzhen.aliyuncs.com/LinBlog/mashibing/basicStructure/class4/stack/9.png)

#### 代码

```java
// 应用
public void sortedArrDistanceLessK(int[] arr, int k) {
    // 默认小根堆
    PriorityQueue<Integer> heap = new PriorityQueue<>();
    int index = 0;
    // min主要是为了 看k有没有比数组长
    // 前k+1个数存入小根堆
    for (; index <= Math.min(arr.length-1, k); index++) {
        // 拷贝入小根堆
        heap.add(arr[index]);
    }
    int i = 0;
    // k+1个数之后的数字 加入小根堆
    for (; index < arr.length; i++, index++) {
        // 先添加 再弹出；先弹出 再添加 也一样
        heap.add(arr[index]);
        // 弹出最小值 默认方法
        arr[i] = heap.poll();
    }
    // 没有值可以加了 把剩余值依次弹出，会从小到大
    while (!heap.isEmpty()) {
        arr[i++] = heap.poll();
    }
}
```

## <a name="2.2">[语言提供的堆结构 vs 手写的堆结构 动态数组实现](#begin)</a>

### 思路

- 取决于，你有没有动态改信息的需求！
- 语言提供的堆结构，如果你动态改数据，不保证依然有序
- 手写堆结构，因为增加了对象的位置表，所以能够满足动态改信息的需求
- 问题：如果根据比较器，堆进行排序后，无法再修改原本插入的元素内容

### 动态数组代码

#### 动态数组

```java
// 堆
public static class MyHeap<T> {
    // 把这个当成动态数组  java中无法实现动态[]
    private ArrayList<T> heap;
    // 记录每个样本在堆上的位置 key为插入的样本
    private HashMap<T, Integer> indexMap;
    // 堆的大小
    private int heapSize;
    // 比较器
    private Comparator<? super T> comparator;

    public MyHeap(Comparator<? super T> com) {
        heap = new ArrayList<>();
        indexMap = new HashMap<>();
        heapSize = 0;
        comparator = com;
    }

    // 堆是否为空
    public boolean isEmpty() {
        return heapSize == 0;
    }

    // 堆大小
    public int size() {
        return heapSize;
    }

    // 是否收过此值
    public boolean contains(T key) {
        return indexMap.containsKey(key);
    }

    // 添加一个数
    public void push(T value) {
        heap.add(value);
        // 对象（不仅仅是值 是内存地址） 位置
        indexMap.put(value, heapSize);
        heapInsert(heapSize++);
    }

    // 弹出一个数 返回的是最小值
    // heap indexMap得同步删除
    public T pop() {
        T ans = heap.get(0);
        int end = heapSize - 1;
        swap(0, end);
        heap.remove(end);
        indexMap.remove(ans);
        heapify(0, --heapSize);
        return ans;
    }

    // 用户说某个value的值变了，但内存地址不变 重新调成大根堆
    public void resign(T value) {
        // 找到在堆上的位置
        int valueIndex = indexMap.get(value);
        // 下面两者必中一个逻辑，要么往上走，要么往下走
        // 跟父节点比大小
        heapInsert(valueIndex);
        // 跟左右节点比大小
        heapify(valueIndex, heapSize);
    }

    // 插入值
    private void heapInsert(int index) {
        // 由比较器 index和父节点比较大小 其余不变
        while (comparator.compare(heap.get(index), heap.get((index - 1) / 2)) < 0) {
            // 跟父进行交换
            swap(index, (index - 1) / 2);
            // index下标变为父节点
            index = (index - 1) / 2;
        }
    }

    // heap indexMap得一起变 保证强同步
    private void swap(int i, int j) {
        T o1 = heap.get(i);
        T o2 = heap.get(j);
        heap.set(i, o2);
        heap.set(j, o1);
        indexMap.put(o1, j);
        indexMap.put(o2, i);
    }

    // 重排
    private void heapify(int index, int heapSize) {
        int left = index * 2 + 1;
        while (left < heapSize) {
            int largest = left + 1 < heapSize && (comparator.compare(heap.get(left + 1), heap.get(left)) < 0) ? left + 1 : left;
            largest = comparator.compare(heap.get(largest), heap.get(index)) < 0 ? largest : index;
            if (largest == index) {
                break;
            }
            swap(largest, index);
            index = largest;
            left = index * 2 + 1;
        }
    }
}
```
